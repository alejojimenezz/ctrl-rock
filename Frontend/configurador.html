iLogic Rule (Autodesk Inventor) 
Imports System.IO
Dim jsonPath As String = "C:\ruta\a\guitar_parameters.json"
If File.Exists(jsonPath) Then
    Dim jsonText As String = File.ReadAllText(jsonPath)
    ' Parsea valores del JSON
    Dim bodyL As Double = Val(jsonText.Split(Chr(34) & "bodyLength" & Chr(34) & ":")(1).Split(",")(0))
    Dim bodyW As Double = Val(jsonText.Split(Chr(34) & "bodyWidth" & Chr(34) & ":")(1).Split(",")(0))
    Dim bodyT As Double = Val(jsonText.Split(Chr(34) & "bodyThickness" & Chr(34) & ":")(1).Split(",")(0))
    Dim scaleL As Double = Val(jsonText.Split(Chr(34) & "scaleLength" & Chr(34) & ":")(1).Split("}")(0))
    
    ' Actualiza parámetros de usuario en Inventor
    Parameter("LargoCuerpo") = bodyL
    Parameter("AnchoCuerpo") = bodyW
    Parameter("EspesorCuerpo") = bodyT
    Parameter("EscalaMastil") = scaleL
    
    RuleParametersOutput()
    InventorVb.DocumentUpdate()
    MessageBox.Show("Parámetros actualizados exitosamente en Inventor.", "Ctrl+Rock CAD")
Else
    MessageBox.Show("Selecciona el archivo descargado.", "Error de iLogic")
End If
